    <p class="hint">（点击上方绿色区域直接复制，号码每秒自动跳动）</p>

    <script>
        function updateOrder() {
            const now = new Date();
            
            // 构造12位纯数字：年(后2位)+月+日+时+分+秒
            // 比如 2026年04月15日 23:58:10 -> 260415235810
            const yy = String(now.getFullYear()).slice(-2);
            const mm = String(now.getMonth() + 1).padStart(2, '0');
            const dd = String(now.getDate()).padStart(2, '0');
            const hh = String(now.getHours()).padStart(2, '0');
            const min = String(now.getMinutes()).padStart(2, '0');
            const ss = String(now.getSeconds()).padStart(2, '0');

            const newOrderId = yy + mm + dd + hh + min + ss;
            document.getElementById('orderNumber').innerText = newOrderId;
        }

        function copyOrder() {
            const num = document.getElementById('orderNumber').innerText;
            navigator.clipboard.writeText(num).then(() => {
                alert('已复制当前秒级的订单号：' + num);
            }).catch(err => {
                alert('复制失败，请手动选择复制');
            });
        }

        // 每1000毫秒（1秒）强制刷新一次数字
        setInterval(updateOrder, 1000);
        
        // 页面打开瞬间先运行一次，避免1秒空白
        updateOrder();
    </script>
